## Usage

```tsx
import { Toggle } from "~/components/ui/toggle"
```

```tsx
<Toggle>
  {(state) => (
    <Show when={state.pressed()} fallback={<Icon2 />}>
      <Icon1 />
    </Show>
  )}
</Toggle>
```
